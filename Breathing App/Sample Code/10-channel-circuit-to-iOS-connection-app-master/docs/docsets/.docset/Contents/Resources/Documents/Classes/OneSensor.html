OneSensor

class OneSensor

This class was designed and implemented to define a StretchSense sensor

Note

The OneSensor is defined by a unique identifier UUID, the present value and an array of the 50 previous values and his color 
// Exemple: Display the sensors connected in a table

   func tableView(tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
       // This fonction give the number of row in the table
       return stretchsenseObject.getNumberOfDeviceConnected()
   }

   func tableView(tableView: UITableView, cellForRowAtIndexPath indexPath: NSIndexPath) -> UITableViewCell {
       // This function is called for each row in the table
       // The row number of the table is define with his indexPath.row

       // The current cell
       let cell = tableView.dequeueReusableCellWithIdentifier("Cell", forIndexPath: indexPath) as UITableViewCell
       // The current peripheral
       let myPeripheral = listPeripheralsConnected[indexPath.row]

       // display the title: value of the sensor
       cell.textLabel?.text = myPeripheralConnected.value
       // display the subtitle: color name of the sensor
       cell.detailTextLabel?.text = myPeripheral.colors[myPeripheral.uniqueNumber].colorName
       // display the background color of the cell: color of the sensor
       cell.backgroundColor = myPeripheral.colors[myPeripheral.uniqueNumber].valueRGB
   }
